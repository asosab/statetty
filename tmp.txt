const mongoose    = require("mongoose");

const inmueblesSchema = new mongoose.Schema(
   {
      id:           {type:String, unique: true},
      brocker:      {type:String},
      URL:          {type:String, unique: true},
      pais:         {type:String},
      ciudad:       {type:String},
      nombre:       {type:String},
      vendida:      {type:Boolean, default:false},
      alquilada:    {type:Boolean, default:false},
      enPlanos:     {type:Boolean},
      nCaptacion:   {type:Boolean, default:false},
      reserva:      {type:Boolean},
      fotos:        {type: Array,"default":[]},
      lat:          {type:String},
      lng:          {type:String},
      videoURL:     {type:String},
      tipoNegocio:  {type:String},
      precio:       {type:Number},
      tipoInmueble: {type:String},
      direccion:    {type:String},
      zona:         {type:String},
      ambientes:    {type:Number},
      dormitorios:  {type:Number},
      estacionamientos:  {type:Number},
      banos:        {type:Number},
      floorLevel:   {type:Number},
      pisos:        {type:Number},
      pileta:       {type:Boolean},
      elevadores:   {type:Number},
      duchas:       {type:Number},
      petsAllowed:  {type:Boolean},
      m2c:          {type:Number},
      m2t:          {type:Number},
      pm2:          {type:Number},
      desc:         {type:String},
      anoc:         {type:Number},
      atributos:    {type: Array,default:[]},
      agenteID:     {type:String},
      agenciaID:    {type:String},
      scrapDetail:  {type:Boolean},
      idDuplicado:  {type:String},                       // El id del registro del que es copia
      activo:       {type:Boolean, default:true},      // representa inmuebles que se están comercializando actualmente
      fecha_ini:    {type:String},
      contrato_fin: {type:String},
      contrato_exclusivo:{type:Boolean, default:true},
      fecha_cancel: {type:String},
      borrador:     {type:Boolean, default:false},
      lastScrapDate:{type: Date},
      error404:     {type:Number},
      agentName:    {type:String},
      agentPhon:    {type:String},
      micros:       {type: Array,default:[]},
      sinMicros:    {type:Boolean, default:false},
      fecha_entrega:{type: Date},
      es_preventa:  {type:Boolean, default:false},
   },{
      timestamps: true,
      versionkey: false
   }
);

inmueblesSchema.index({ desc: 1 }, { collation: { locale: 'es', strength: 1 } });
inmueblesSchema.index({ activo: 1, createdAt: 1 });
inmueblesSchema.index({ activo: 1, createdAt: 1, precio: 1 });
inmueblesSchema.index({ activo: 1, createdAt: 1, ciudad: 1 });
inmueblesSchema.index({ agenteID: 1 });
inmueblesSchema.index({ agenciaID: 1 });

module.exports = mongoose.model('inmuebles', inmueblesSchema);
